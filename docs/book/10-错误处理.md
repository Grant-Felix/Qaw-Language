## 10.1 Result 与 Option

```yao
func read_file(path: string) -> Result<string, io::Error> {
    let content = std.fs.read_to_string(path)?;   // ? 传播错误
    Ok(content)
}

let first = first_word("hello world");
match first {
    Some(s) => print(s),
    None => print("无词"),
}
```

## 10.2 ? 运算符

```yao
func process() -> Result<int, MyError> {
    let x = step1()?;      // Err 时立即返回
    let y = step2(x)?;
    Ok(y)
}
```

## 10.3 自定义错误

```yao
trait Error: Display + Debug {
    method source(self) -> Option<&dyn Error>;
}

struct MyError {
    message: string,
    source: Option<Box<dyn Error>>,
}

impl Display for MyError {
    method fmt(self, f: &mut Formatter) -> Result {
        write!(f, "MyError: {}", self.message)
    }
}

impl Error for MyError {
    method source(self) -> Option<&dyn Error> {
        self.source.as_ref().map(|e| e.as_ref())
    }
}
```
