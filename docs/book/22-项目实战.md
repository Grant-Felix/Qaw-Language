## 22.1 项目 1：Hello 妖文与四则运算计算器

**难度**：★☆☆ | **涉及**：基础语法、词法、表达式

```yao
package calc;

func calc(a: int, b: int, op: char) -> Result<int, string> {
    match op {
        '+' => Ok(a + b),
        '-' => Ok(a - b),
        '*' => Ok(a * b),
        '/' => {
            if b == 0 { return Err("除数不能为零"); }
            Ok(a / b)
        }
        _ => Err("未知运算符"),
    }
}

func main() -> void {
    print("四则运算计算器（输入 'q' 退出）");
    
    loop {
        print("> ");
        let line = read_line().unwrap_or_default();
        
        if line.trim() == "q" { break; }
        if line.trim().is_empty() { continue; }
        
        let parts: [string] = line.split_whitespace().collect();
        if parts.len() != 3 { continue; }
        
        let a = parts[0].parse::<int>().unwrap_or(0);
        let b = parts[2].parse::<int>().unwrap_or(0);
        let op = parts[1].chars().next().unwrap();
        
        match calc(a, b, op) {
            Ok(r) => print("${a} ${op} ${b} = ${r}"),
            Err(e) => print("错误: ${e}"),
        }
    }
}
```

## 22.2 项目 2：文件批处理工具

**难度**：★★☆ | **涉及**：包、模块、错误处理、文件 IO

支持批量重命名、文本替换、行数统计。

```yao
package file_batch;

use std.fs;
use std.path::Path;

enum Op {
    Rename { from: string, to: string },
    Replace { old: string, new: string },
    Count,
}

fn process(op: Op, paths: [string]) -> Result<()> {
    for path in paths {
        let files = list_files(&path, false)?;
        for file in files {
            match &op {
                Op::Rename { from, to } => {
                    let new_name = file.replace(from, to);
                    fs::rename(&file, &new_name)?;
                }
                Op::Replace { old, new } => {
                    let content = fs::read_to_string(&file)?;
                    fs::write(&file, &content.replace(old, new))?;
                }
                Op::Count => {
                    let content = fs::read_to_string(&file)?;
                    println("${file}: ${content.lines().count()} 行");
                }
            }
        }
    }
    Ok(())
}
```

## 22.3 项目 3：并发爬虫

**难度**：★★☆ | **涉及**：协程、通道、select、超时

```yao
package crawler;

use std.http;
use std.runtime;

func crawl(urls: [string]) -> [string] {
    let (tx, rx) = make(chan<string>, 100);
    
    for url in urls {
        let tx = tx.clone();
        go async {
            let resp = http.get(url).await?;
            let title = extract_title(&resp.text().await?);
            tx <- title;
        };
    }
    
    drop(tx);  // 关闭发送端
    
    let mut results = [];
    for title in rx {
        results.push(title);
    }
    results
}
```

## 22.4 项目 4：表达式求值器

**难度**：★★★ | **涉及**：递归下降解析器、ADT、trait、闭包

支持 `+ - * / %`、`let x = 5 in expr`、`fn x => expr`。

```yao
package expr_eval;

enum Expr {
    Int(i64),
    Var(string),
    Binary { op: char, left: Box<Expr>, right: Box<Expr> },
    Let { name: string, value: Box<Expr>, body: Box<Expr> },
    Lambda { param: string, body: Box<Expr> },
    Call { func: Box<Expr>, arg: Box<Expr> },
}

enum Value {
    IntVal(i64),
    Closure { param: string, body: Box<Expr>, env: Env },
}

fn eval(expr: &Expr, env: &Env) -> Result<Value, string> {
    match expr {
        Expr::Int(n) => Ok(Value::IntVal(*n)),
        Expr::Var(name) => env.get(name).ok_or_else(|| format("未定义变量: ${name}")),
        Expr::Binary { op, left, right } => {
            let l = eval(left, env)?;
            let r = eval(right, env)?;
            apply_binary(*op, l, r)
        }
        Expr::Let { name, value, body } => {
            let v = eval(value, env)?;
            let new_env = env.child();
            new_env.define(name.clone(), v);
            eval(body, &new_env)
        }
        Expr::Lambda { param, body } => {
            Ok(Value::Closure { param: param.clone(), body: body.clone(), env: env.clone() })
        }
        Expr::Call { func, arg } => {
            let f = eval(func, env)?;
            let a = eval(arg, env)?;
            match f {
                Value::Closure { param, body, env } => {
                    let new_env = env.child();
                    new_env.define(param, a);
                    eval(&body, &new_env)
                }
                _ => Err("非函数调用".to_string()),
            }
        }
    }
}
```

## 22.5 项目 5：HTTP 服务器

**难度**：★★★ | **涉及**：异步 IO、协程、trait、trait object

```yao
package http_server;

use std.net::TcpListener;

trait Handler {
    async method call(self, req: Request) -> Response;
}

struct Router {
    routes: [(string, string, Box<dyn Handler>)],
}

impl Router {
    async method handle(self, req: Request) -> Response {
        for (method, path, handler) in &self.routes {
            if method == &req.method && path == &req.path {
                return handler.call(req).await;
            }
        }
        Response::new("404 Not Found")
    }
}

async func main() -> void {
    let listener = TcpListener::bind("127.0.0.1:8080").await?;
    println("HTTP 服务器运行于 http://127.0.0.1:8080");
    
    loop {
        let (stream, _) = listener.accept().await?;
        go async {
            handle_connection(stream).await;
        };
    }
}
```

## 22.6 项目 6：LRU 缓存

**难度**：★★★ | **涉及**：@shared、Arc、Mutex、Ref

```yao
package lru;

use std.collections::HashMap;
use std.sync::{Arc, Mutex};

@shared
struct LruCache<K, V> {
    capacity: usize,
    map: HashMap<K, V>,
    order: VecDeque<K>,
}

impl<K: Hash + Eq + Clone, V: Clone> LruCache<K, V> {
    method new(capacity: usize) -> Self {
        assert!(capacity > 0);
        LruCache {
            capacity,
            map: HashMap::new(),
            order: VecDeque::new(),
        }
    }
    
    method get(mut self, key: &K) -> Option<V> {
        if !self.map.contains(key) { return None; }
        self.order.retain(|k| k != key);
        self.order.push_front(key.clone());
        self.map.get(key).cloned()
    }
    
    method put(mut self, key: K, value: V) {
        if self.map.contains(&key) {
            self.order.retain(|k| k != key);
        } else if self.order.len() >= self.capacity {
            if let Some(oldest) = self.order.pop_back() {
                self.map.remove(&oldest);
            }
        }
        self.order.push_front(key.clone());
        self.map.insert(key, value);
    }
}
```

## 22.7 项目 7：JSON 解析器

**难度**：★★★ | **涉及**：手写解析器、trait、错误处理、宏

```yao
package json;

enum JsonValue {
    Null,
    Bool(bool),
    Number(f64),
    String(string),
    Array(Vec<JsonValue>),
    Object(HashMap<string, JsonValue>),
}

trait Deserialize: Sized {
    fn deserialize(value: &JsonValue) -> Result<Self, string>;
}

#[derive(Deserialize)]
struct User {
    id: i64,
    name: string,
    email: Option<string>,
}

fn main() -> void {
    let json_str = r#"{"id": 42, "name": "张三"}"#;
    let value = parse(json_str).unwrap();
    let user: User = User::deserialize(&value).unwrap();
    println("${user.name}");
}
```
